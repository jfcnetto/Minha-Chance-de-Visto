        // Sistema simples de troca de abas (Single Page Application sem frameworks ou IA)
        function switchPage(pageId) {
            // Oculta todas as seções
            const sections = document.querySelectorAll('.page-section');
            sections.forEach(section => section.classList.remove('active'));
            
            // Remove a classe ativa de todos os links de navegação
            const navLinks = document.querySelectorAll('nav a');
            navLinks.forEach(link => link.classList.remove('active'));

            // Mostra a seção desejada
            document.getElementById(pageId).classList.add('active');
            
            // Ativa o link correspondente se ele estiver no menu principal
            const activeNav = document.getElementById('nav-' + pageId);
            if (activeNav) activeNav.classList.add('active');

            // Rola para o topo de forma suave
            window.scrollTo({top: 0, behavior: 'smooth'});
        }

        // Lógica matemática pura para o cálculo de chances
        function calculateVisaChance(event) {
            event.preventDefault();

            const visaType = document.getElementById('visaType').value;
            const vinculos = document.getElementById('vinculos').value;
            const financeiro = document.getElementById('financeiro').value;
            const historico = document.getElementById('historico').value;

            let score = 0;

            // Pontuação por Vínculos (Fator Crítico na Lei de Imigração dos EUA)
            if (vinculos === 'fortes') score += 40;
            else if (vinculos === 'medios') score += 20;
            else score += 5;

            // Pontuação por Estabilidade Financeira
            if (financeiro === 'excelente') score += 30;
            else if (financeiro === 'moderada') score += 15;
            else score += 0;

            // Pontuação por Histórico de Viagens
            if (historico === 'sim') score += 30;
            else if (historico === 'nao') score += 15;
            else score -= 20; // Penalidade por problemas anteriores

            // Ajuste fino dependendo se for Green Card (Critérios mais rigorosos de elegibilidade)
            if (visaType === 'greencard') {
                score = score * 0.85; // Reduz a probabilidade bruta para refletir a dificuldade real
            }

            // Exibição do Resultado
            const resultBox = document.getElementById('result-box');
            const resultTitle = document.getElementById('result-title');
            const resultText = document.getElementById('result-text');

            resultBox.className = ''; // Limpa classes antigas
            resultBox.style.display = 'block';

            if (score >= 70) {
                resultBox.classList.add('chance-alta');
                resultTitle.innerText = "Chance Estimada: Alta (" + Math.round(score) + "%)";
                resultText.innerText = "Seu perfil apresenta ótimos indicadores de aprovação. Certifique-se de manter todos os documentos atualizados, organizados e claros para o dia da sua entrevista.";
            } else if (score >= 40) {
                resultBox.classList.add('chance-media');
                resultTitle.innerText = "Chance Estimada: Moderada (" + Math.round(score) + "%)";
                resultText.innerText = "Seu perfil tem boas chances, mas possui pontos que merecem atenção. Procure focar em demonstrar mais vínculos e clareza financeira no momento do preenchimento da DS-160.";
            } else {
                resultBox.classList.add('chance-baixa');
                resultTitle.innerText = "Chance Estimada: Atenção Requerida (" + Math.max(10, Math.round(score)) + "%)";
                resultText.innerText = "Seus vínculos ou histórico atual podem levantar dúvidas junto ao consulado. Recomendamos que melhore seus laços de emprego, estudos ou situação financeira antes de realizar a aplicação.";
            }

            // Desloca a tela para o resultado
            resultBox.scrollIntoView({ behavior: 'smooth' });
        }
    </script>

📊 1. O Nível de Concorrência neste Nicho
O nicho de "Visto Americano" e "Imigração/Green Card" é considerado de concorrência média-alta para alta, porém com uma característica muito específica: alto valor financeiro (CPC elevado).

Quem são os concorrentes grandes? Grandes escritórios de advocacia imigratória e agências de assessoria de visto. Eles investem pesado em anúncios pagos (Google Ads) e possuem blogs com forte autoridade.

Onde está a brecha para você? A maioria desses sites grandes não quer monetizar com AdSense; eles querem vender uma assessoria que custa de R$ 500 a mais de R$ 30.000 (no caso de Green Cards). Eles veem o usuário que quer apenas uma informação gratuita ou simulação como "tráfego frio". É exatamente aí que o seu MVP entra: capturando a massa de usuários que quer resolver ou entender o processo por conta própria.

Dificuldade de SEO: Palavras-chave genéricas como "como tirar visto americano" são extremamente difíceis de ranquear no topo logo de início. No entanto, termos de cauda longa (ex: "endereço consulado americano porto alegre" ou "simulador de chances de visto") oferecem uma concorrência muito menor e mais fácil de dominar.
